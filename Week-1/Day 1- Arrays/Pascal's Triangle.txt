class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>>result;
        for(int row=1;row<=numRows;row++){
            vector<int>temp;
            long long ans=1;
            temp.push_back(ans);
            for(int col=1;col<row;col++){
                ans=ans*(row-col);
                ans=ans/col;
                temp.push_back(ans);
            }
            result.push_back(temp);    
        }
        return result;
    }
};
